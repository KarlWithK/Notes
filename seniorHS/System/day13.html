	// Bit wise negation -> ~
	a = ~a
	// bitwise or -> think geometry
	a | b
	// bitwise and -> think geometry
	a & b
	// bitwise xor -> think geometry. Only one has to be true and the other has to be false
	a & b
	// shift operators
	// right shift by n bits -> add zeros at the start
	a >> n
	// left shift by n bits -> add zeros at the end
	a << n
	// value swap bitwise		
	a = a ^ b // unique bits not shared by a or b
	b = a ^ b
	a = a ^ b
